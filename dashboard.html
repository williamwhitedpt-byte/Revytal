Supabase Schema & Setup Reference

1 — Run this SQL in your Supabase SQL Editor

-- ── clinics ───────────────────────────────────────────────── create table clinics ( id uuid primary key default gen_random_uuid(), name text not null, slug text unique, contact_email text, created_at timestamptz default now() ); -- Insert Mend Physical Therapy (copy the returned id into CLINIC_ID) insert into clinics (name, slug, contact_email) values ('Mend Physical Therapy', 'mend-pt', 'will@mendtherapyla.com') returning id; -- ── products ───────────────────────────────────────────────── create table products ( id uuid primary key default gen_random_uuid(), product_name text not null, sku text unique not null, supplier text not null, price numeric(10,2) not null, cost numeric(10,2), url text, tags text, active boolean default true, created_at timestamptz default now() ); -- ── orders ─────────────────────────────────────────────────── -- PHI NOTE: patient_initials only (e.g. 'J.D.'). No full names, DOB, -- phone numbers, emails, or any other patient identifiers. create table orders ( id uuid primary key default gen_random_uuid(), clinic_id uuid references clinics(id), product_name text not null, sku text not null, supplier text not null, price numeric(10,2) not null, cost numeric(10,2), margin numeric(10,2) generated always as (price - coalesce(cost,0)) stored, patient_initials text, -- 'J.D.' — NO other patient data checkout_url text, stripe_session_id text unique, status text default 'pending' check (status in ('pending','paid','fulfilled','failed')), created_at timestamptz default now(), fulfilled_at timestamptz ); -- Indexes create index on orders (clinic_id); create index on orders (stripe_session_id); create index on orders (created_at desc); create index on orders (status); -- Enable realtime (run in Supabase Dashboard → Database → Replication) -- Or run: alter publication supabase_realtime add table orders;

3 — Stripe Webhook (Next.js / Node handler pseudocode)

// Listen for: checkout.session.completed // Stripe Dashboard → Developers → Webhooks → Add endpoint // URL: https://your-backend.com/api/webhook const event = stripe.webhooks.constructEvent(body, sig, secret); if (event.type === 'checkout.session.completed') { const session = event.data.object; const { sku, product_name, supplier, price, cost, clinic_id, patient_initials } = session.metadata; // ↑ Pass these as metadata when creating the Checkout Session // patient_initials = initials only — never full name await supabase.from('orders').insert({ clinic_id, product_name, sku, supplier, price: parseFloat(price), cost: parseFloat(cost), patient_initials, // e.g. 'J.D.' checkout_url: session.url, stripe_session_id: session.id, status: 'paid' }); // Then send supplier email (see main app) await routeToSupplier({ sku, supplier, patient_initials }); // Update to fulfilled after supplier routing await supabase.from('orders') .update({ status: 'fulfilled', fulfilled_at: new Date() }) .eq('stripe_session_id', session.id); }

4 — Row Level Security (recommended)

alter table orders enable row level security; -- Allow service role full access (webhook uses this) create policy "service_role_all" on orders as permissive for all to service_role using (true) with check (true); -- Allow anon read for dashboard (scope to clinic_id) create policy "anon_read_clinic" on orders for select to anon using (clinic_id = 'YOUR_CLINIC_ID'::uuid);
